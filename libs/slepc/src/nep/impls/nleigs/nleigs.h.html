 54: PETSC_STATIC_INLINE PetscErrorCode NEPNLEIGSSetShifts(NEP nep,PetscInt *nshiftsw)
 55: {
 56:   NEP_NLEIGS *ctx = (NEP_NLEIGS*)nep->data;

 59:   if (!ctx->nshifts) {
 60:     ctx->shifts = &nep->target;
 61:     *nshiftsw = 1;
 62:   } else *nshiftsw = ctx->nshifts;
 63:   return(0);
 64: }
