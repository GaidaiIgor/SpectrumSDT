 22:    Not Collective

 24:    Level: advanced

 28: PetscErrorCode BVRegisterAll(void)
 29: {

 37:   BVRegister(BVSVEC,BVCreate_Svec);

 39:   BVRegister(BVTENSOR,BVCreate_Tensor);

 41: }
