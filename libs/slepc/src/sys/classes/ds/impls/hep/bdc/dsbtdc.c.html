 12: */

 14:  #include <slepc/private/dsimpl.h>
 15:  #include <slepcblaslapack.h>

 17: PetscErrorCode BDC_dsbtdc_(const char *jobz,const char *jobacc,PetscBLASInt n,
 18:         PetscBLASInt nblks,PetscBLASInt *ksizes,PetscReal *d,PetscBLASInt l1d,
 19:         PetscBLASInt l2d,PetscReal *e,PetscBLASInt l1e,PetscBLASInt l2e,PetscReal tol,
 20:         PetscReal tau1,PetscReal tau2,PetscReal *ev,PetscReal *z,PetscBLASInt ldz,
 21:         PetscReal *work,PetscBLASInt lwork,PetscBLASInt *iwork,PetscBLASInt liwork,
 22:         PetscReal *mingap,PetscBLASInt *mingapi,PetscBLASInt *info,
 23:         PetscBLASInt jobz_len,PetscBLASInt jobacc_len)
 24: {
 25: /*  -- Routine written in LAPACK Version 3.0 style -- */
 26: /* *************************************************** */
 27: /*     Written by */
 28: /*     Michael Moldaschl and Wilfried Gansterer */
 29: /*     University of Vienna */
 30: /*     last modification: March 28, 2014 */

 32: /*     Small adaptations of original code written by */
 33: /*     Wilfried Gansterer and Bob Ward, */
 34: /*     Department of Computer Science, University of Tennessee */
 35: /*     see https://doi.org/10.1137/S1064827501399432 */
 36: /* *************************************************** */

 38: /*  Purpose */
 39: /*  ======= */

 41: /*  DSBTDC computes approximations to all eigenvalues and eigenvectors */
 42: /*  of a symmetric block tridiagonal matrix using the divide and */
 43: /*  conquer method with lower rank approximations to the subdiagonal blocks. */

 45: /*  This code makes very mild assumptions about floating point */
 46: /*  arithmetic. It will work on machines with a guard digit in */
 47: /*  add/subtract, or on those binary machines without guard digits */
 48: /*  which subtract like the Cray X-MP, Cray Y-MP, Cray C-90, or Cray-2. */
 49: /*  It could conceivably fail on hexadecimal or decimal machines */
 50: /*  without guard digits, but we know of none.  See DLAED3M for details. */

 52: /*  Arguments */
 53: /*  ========= */

 55: /*  JOBZ    (input) CHARACTER*1 */
 56: /*          = 'N':  Compute eigenvalues only (not implemented); */
 57: /*          = 'D':  Compute eigenvalues and eigenvectors. Eigenvectors */
 58: /*                  are accumulated in the divide-and-conquer process. */

 60: /*  JOBACC  (input) CHARACTER*1 */
 61: /*          = 'A' ("automatic"): The accuracy parameters TAU1 and TAU2 */
 62: /*                               are determined automatically from the */
 63: /*                               parameter TOL according to the analytical */
 64: /*                               bounds. In that case the input values of */
 65: /*                               TAU1 and TAU2 are irrelevant (ignored). */
 66: /*          = 'M' ("manual"): The input values of the accuracy parameters */
 67: /*                            TAU1 and TAU2 are used. In that case the input */
 68: /*                            value of the parameter TOL is irrelevant */
 69: /*                            (ignored). */

 71: /*  N       (input) INTEGER */
 72: /*          The dimension of the symmetric block tridiagonal matrix. */
 73: /*          N >= 1. */

 75: /*  NBLKS   (input) INTEGER, 1 <= NBLKS <= N */
 76: /*          The number of diagonal blocks in the matrix. */

 78: /*  KSIZES  (input) INTEGER array, dimension (NBLKS) */
 79: /*          The dimensions of the square diagonal blocks from top left */
 80: /*          to bottom right.  KSIZES(I) >= 1 for all I, and the sum of */
 81: /*          KSIZES(I) for I = 1 to NBLKS has to be equal to N. */

 83: /*  D       (input) DOUBLE PRECISION array, dimension (L1D,L2D,NBLKS) */
 84: /*          The lower triangular elements of the symmetric diagonal */
 85: /*          blocks of the block tridiagonal matrix. The elements of the top */
 86: /*          left diagonal block, which is of dimension KSIZES(1), have to */
 87: /*          be placed in D(*,*,1); the elements of the next diagonal */
 88: /*          block, which is of dimension KSIZES(2), have to be placed in */
 89: /*          D(*,*,2); etc. */

 91: /*  L1D     (input) INTEGER */
 92: /*          The leading dimension of the array D.  L1D >= max(3,KMAX), */
 93: /*          where KMAX is the dimension of the largest diagonal block, */
 94: /*          i.e.,  KMAX = max_I (KSIZES(I)). */

 96: /*  L2D     (input) INTEGER */
 97: /*          The second dimension of the array D.  L2D >= max(3,KMAX), */
 98: /*          where KMAX is as stated in L1D above. */

100: /*  E       (input) DOUBLE PRECISION array, dimension (L1E,L2E,NBLKS-1) */
101: /*          The elements of the subdiagonal blocks of the */
102: /*          block tridiagonal matrix. The elements of the top left */
103: /*          subdiagonal block, which is KSIZES(2) x KSIZES(1), have to be */
104: /*          placed in E(*,*,1); the elements of the next subdiagonal block, */
105: /*          which is KSIZES(3) x KSIZES(2), have to be placed in E(*,*,2); etc. */
106: /*          During runtime, the original contents of E(*,*,K) is */
107: /*          overwritten by the singular vectors and singular values of */
108: /*          the lower rank representation. */

110: /*  L1E     (input) INTEGER */
111: /*          The leading dimension of the array E.  L1E >= max(3,2*KMAX+1), */
112: /*          where KMAX is as stated in L1D above. The size of L1E enables */
113: /*          the storage of ALL singular vectors and singular values for */
114: /*          the corresponding off-diagonal block in E(*,*,K) and therefore */
115: /*          there are no restrictions on the rank of the approximation */
116: /*          (only the "natural" restriction */
117: /*          RANK(K) .LE. MIN(KSIZES(K),KSIZES(K+1))). */

119: /*  L2E     (input) INTEGER */
120: /*          The second dimension of the array E.  L2E >= max(3,2*KMAX+1), */
121: /*          where KMAX is as stated in L1D above. The size of L2E enables */
122: /*          the storage of ALL singular vectors and singular values for */
123: /*          the corresponding off-diagonal block in E(*,*,K) and therefore */
124: /*          there are no restrictions on the rank of the approximation */
125: /*          (only the "natural" restriction */
126: /*          RANK(K) .LE. MIN(KSIZES(K),KSIZES(K+1))). */

128: /*  TOL     (input) DOUBLE PRECISION, TOL.LE.TOLMAX */
129: /*          User specified tolerance for the residuals of the computed */
130: /*          eigenpairs. If (JOBACC.EQ.'A') then it is used to determine */
131: /*          TAU1 and TAU2; ignored otherwise. */
132: /*          If (TOL.LT.40*EPS .AND. JOBACC.EQ.'A') then TAU1 is set to machine */
133: /*          epsilon and TAU2 is set to the standard deflation tolerance from */
134: /*          LAPACK. */

136: /*  TAU1    (input) DOUBLE PRECISION, TAU1.LE.TOLMAX/2 */
137: /*          User specified tolerance for determining the rank of the */
138: /*          lower rank approximations to the off-diagonal blocks. */
139: /*          The rank for each off-diagonal block is determined such that */
140: /*          the resulting absolute eigenvalue error is less than or equal */
141: /*          to TAU1. */
142: /*          If (JOBACC.EQ.'A') then TAU1 is determined automatically from */
143: /*             TOL and the input value is ignored. */
144: /*          If (JOBACC.EQ.'M' .AND. TAU1.LT.20*EPS) then TAU1 is set to */
145: /*             machine epsilon. */

147: /*  TAU2    (input) DOUBLE PRECISION, TAU2.LE.TOLMAX/2 */
148: /*          User specified deflation tolerance for the routine DIBTDC. */
149: /*          If (1.0D-1.GT.TAU2.GT.20*EPS) then TAU2 is used as */
150: /*          the deflation tolerance in DSRTDF (EPS is the machine epsilon). */
151: /*          If (TAU2.LE.20*EPS) then the standard deflation tolerance from */
152: /*          LAPACK is used as the deflation tolerance in DSRTDF. */
153: /*          If (JOBACC.EQ.'A') then TAU2 is determined automatically from */
154: /*             TOL and the input value is ignored. */
155: /*          If (JOBACC.EQ.'M' .AND. TAU2.LT.20*EPS) then TAU2 is set to */
156: /*             the standard deflation tolerance from LAPACK. */

158: /*  EV      (output) DOUBLE PRECISION array, dimension (N) */
159: /*          If INFO = 0, then EV contains the computed eigenvalues of the */
160: /*          symmetric block tridiagonal matrix in ascending order. */

162: /*  Z       (output) DOUBLE PRECISION array, dimension (LDZ,N) */
163: /*          If (JOBZ.EQ.'D' .AND. INFO = 0) */
164: /*          then Z contains the orthonormal eigenvectors of the symmetric */
165: /*          block tridiagonal matrix computed by the routine DIBTDC */
166: /*          (accumulated in the divide-and-conquer process). */
167: /*          If (-199 < INFO < -99) then Z contains the orthonormal */
168: /*          eigenvectors of the symmetric block tridiagonal matrix, */
169: /*          computed without divide-and-conquer (quick returns). */
170: /*          Otherwise not referenced. */

172: /*  LDZ     (input) INTEGER */
173: /*          The leading dimension of the array Z.  LDZ >= max(1,N). */

175: /*  WORK    (workspace/output) DOUBLE PRECISION array, dimension (LWORK) */

177: /*  LWORK   (input) INTEGER */
178: /*          The dimension of the array WORK. */
179: /*          If NBLKS.EQ.1, then LWORK has to be at least 2N^2+6N+1 */
180: /*          (for the call of DSYEVD). */
181: /*          If NBLKS.GE.2 and (JOBZ.EQ.'D') then the absolute minimum */
182: /*             required for DIBTDC is (N**2 + 3*N). This will not always */
183: /*             suffice, though, the routine will return a corresponding */
184: /*             error code and report how much work space was missing (see */
185: /*             INFO). */
186: /*          In order to guarantee correct results in all cases where */
187: /*          NBLKS.GE.2, LWORK must be at least (2*N**2 + 3*N). */

189: /*  IWORK   (workspace/output) INTEGER array, dimension (LIWORK) */

191: /*  LIWORK  (input) INTEGER */
192: /*          The dimension of the array IWORK. */
193: /*          LIWORK must be at least (5*N + 5*NBLKS - 1) (for DIBTDC) */
194: /*          Note that this should also suffice for the call of DSYEVD on a */
195: /*          diagonal block which requires (5*KMAX + 3). */


198: /*  MINGAP  (output) DOUBLE PRECISION */
199: /*          The minimum "gap" between the approximate eigenvalues */
200: /*          computed, i.e., MIN( ABS(EV(I+1)-EV(I)) for I=1,2,..., N-1 */
201: /*          IF (MINGAP.LE.TOL/10) THEN a warning flag is returned in INFO, */
202: /*          because the computed eigenvectors may be unreliable individually */
203: /*          (only the subspaces spanned are approximated reliably). */

205: /*  MINGAPI (output) INTEGER */
206: /*          Index I where the minimum gap in the spectrum occurred. */

208: /*  INFO    (output) INTEGER */
209: /*          = 0:  successful exit, no special cases occurred. */
210: /*          < -200: not enough workspace. Space for ABS(INFO + 200) */
211: /*                numbers is required in addition to the workspace provided, */
212: /*                otherwise some of the computed eigenvectors will be incorrect. */
213: /*          < -99, > -199: successful exit, but quick returns. */
214: /*                if INFO = -100, successful exit, but the input matrix */
215: /*                                was the zero matrix and no */
216: /*                                divide-and-conquer was performed */
217: /*                if INFO = -101, successful exit, but N was 1 and no */
218: /*                                divide-and-conquer was performed */
219: /*                if INFO = -102, successful exit, but only a single */
220: /*                                dense block. Standard dense solver */
221: /*                                was called, no divide-and-conquer was */
222: /*                                performed */
223: /*                if INFO = -103, successful exit, but warning that */
224: /*                                MINGAP.LE.TOL/10 and therefore the */
225: /*                                eigenvectors corresponding to close */
226: /*                                approximate eigenvalues may individually */
227: /*                                be unreliable (although taken together they */
228: /*                                do approximate the corresponding subspace to */
229: /*                                the desired accuracy) */
230: /*          = -99: error in the preprocessing in DIBTDC (when determining */
231: /*                 the merging order). */
232: /*          < 0, > -99: illegal arguments. */
233: /*                if INFO = -i, the i-th argument had an illegal value. */
234: /*          > 0:  The algorithm failed to compute an eigenvalue while */
235: /*                working on the submatrix lying in rows and columns */
236: /*                INFO/(N+1) through mod(INFO,N+1). */

238: /*  Further Details */
239: /*  =============== */

241: /*  Small modifications of code written by */
242: /*     Wilfried Gansterer and Bob Ward, */
243: /*     Department of Computer Science, University of Tennessee */
244: /*     see https://doi.org/10.1137/S1064827501399432 */

246: /*  Based on the design of the LAPACK code sstedc.f written by Jeff */
247: /*  Rutter, Computer Science Division, University of California at */
248: /*  Berkeley, and modified by Francoise Tisseur, University of Tennessee. */

250: /*  ===================================================================== */

252: /*     .. Parameters .. */

254: #define TOLMAX 0.1

256: /*        TOLMAX       .... upper bound for tolerances TOL, TAU1, TAU2 */
257: /*                          NOTE: in the routine DIBTDC, the value */
258: /*                                1.D-1 is hardcoded for TOLMAX ! */

260:   PetscBLASInt   i, j, k, i1, iwspc, lwmin, start;
261:   PetscBLASInt   ii, ip, nk, rk, np, iu, rp1, ldu;
262:   PetscBLASInt   ksk, ivt, iend, kchk=0, kmax=0, one=1, zero=0;
263:   PetscBLASInt   ldvt, ksum=0, kskp1, spneed, nrblks, liwmin, isvals;
264:   PetscReal      p, d2, eps, dmax, emax, done = 1.0;
265:   PetscReal      dnrm, tiny, anorm, exdnrm=0, dropsv, absdiff;

269:   /* Determine machine epsilon. */
270:   eps = LAPACKlamch_("Epsilon");

272:   *info = 0;

274:   if (*(unsigned char *)jobz != 'N' && *(unsigned char *)jobz != 'D') *info = -1;
275:   else if (*(unsigned char *)jobacc != 'A' && *(unsigned char *)jobacc != 'M') *info = -2;
276:   else if (n < 1) *info = -3;
277:   else if (nblks < 1 || nblks > n) *info = -4;
278:   if (*info == 0) {
279:     for (k = 0; k < nblks; ++k) {
280:       ksk = ksizes[k];
281:       ksum += ksk;
282:       if (ksk > kmax) kmax = ksk;
283:       if (ksk < 1) kchk = 1;
284:     }
285:     if (nblks == 1) lwmin = 2*n*n + n*6 + 1;
286:     else lwmin = n*n + n*3;
287:     liwmin = n * 5 + nblks * 5 - 4;
288:     if (ksum != n || kchk == 1) *info = -5;
289:     else if (l1d < PetscMax(3,kmax)) *info = -7;
290:     else if (l2d < PetscMax(3,kmax)) *info = -8;
291:     else if (l1e < PetscMax(3,2*kmax+1)) *info = -10;
292:     else if (l2e < PetscMax(3,2*kmax+1)) *info = -11;
293:     else if (*(unsigned char *)jobacc == 'A' && tol > TOLMAX) *info = -12;
294:     else if (*(unsigned char *)jobacc == 'M' && tau1 > TOLMAX/2) *info = -13;
295:     else if (*(unsigned char *)jobacc == 'M' && tau2 > TOLMAX/2) *info = -14;
296:     else if (ldz < PetscMax(1,n)) *info = -17;
297:     else if (lwork < lwmin) *info = -19;
298:     else if (liwork < liwmin) *info = -21;
299:   }

301:   if (*info) SETERRQ1(PETSC_COMM_SELF,PETSC_ERR_ARG_WRONG,"Wrong argument %d in DSBTDC",-(*info));

303:   /* Quick return if possible */

305:   if (n == 1) {
306:     ev[0] = d[0]; z[0] = 1.;
307:     *info = -101;
308:     return(0);
309:   }

311:   /* If NBLKS is equal to 1, then solve the problem with standard */
312:   /* dense solver (in this case KSIZES(1) = N). */

314:   if (nblks == 1) {
315:     for (i = 0; i < n; ++i) {
316:       for (j = 0; j <= i; ++j) {
317:         z[i + j*ldz] = d[i + j*l1d];
318:       }
319:     }
320:     PetscStackCallBLAS("LAPACKsyevd",LAPACKsyevd_("V", "L", &n, z, &ldz, ev, work, &lwork, iwork, &liwork, info));
321:     if (*info) SETERRQ1(PETSC_COMM_SELF,1,"dsbtdc: Error in DSYEVD, info = %d",*info);
322:     *info = -102;
323:     return(0);
324:   }

326:   /* determine the accuracy parameters (if requested) */

328:   if (*(unsigned char *)jobacc == 'A') {
329:     tau1 = tol / 2;
330:     if (tau1 < eps * 20) tau1 = eps;
331:     tau2 = tol / 2;
332:   }

334:   /* Initialize Z as the identity matrix */

336:   if (*(unsigned char *)jobz == 'D') {
337:     for (j=0;j<n;j++) for (i=0;i<n;i++) z[i+j*ldz] = 0.0;
338:     for (i=0;i<n;i++) z[i+i*ldz] = 1.0;
339:   }

341:   /* Determine the off-diagonal ranks, form and store the lower rank */
342:   /* approximations based on the tolerance parameters, the */
343:   /* RANK(K) largest singular values and the associated singular */
344:   /* vectors of each subdiagonal block. Also find the maximum norm of */
345:   /* the subdiagonal blocks (in EMAX). */

347:   /* Compute SVDs of the subdiagonal blocks.... */

349:   /* EMAX .... maximum norm of the off-diagonal blocks */

351:   emax = 0.;
352:   for (k = 0; k < nblks-1; ++k) {
353:     ksk = ksizes[k];
354:     kskp1 = ksizes[k+1];
355:     isvals = 0;

357:     /* Note that min(KSKP1,KSK).LE.N/2 (equality possible for */
358:     /* NBLKS=2), and therefore storing the singular values requires */
359:     /* at most N/2 entries of the *        array WORK. */

361:     iu = isvals + n / 2;
362:     ivt = isvals + n / 2;

364:     /* Call of DGESVD: The space for U is not referenced, since */
365:     /* JOBU='O' and therefore this portion of the array WORK */
366:     /* is not referenced for U. */

368:     ldu = kskp1;
369:     ldvt = PetscMin(kskp1,ksk);
370:     iwspc = ivt + n * n / 2;

372:     /* Note that the minimum workspace required for this call */
373:     /* of DGESVD is: N/2 for storing the singular values + N**2/2 for */
374:     /* storing V^T + 5*N/2 workspace =  N**2/2 + 3*N. */

376:     i1 = lwork - iwspc;
377:     PetscStackCallBLAS("LAPACKgesvd",LAPACKgesvd_("O", "S", &kskp1, &ksk,
378:             &e[k*l1e*l2e], &l1e, &work[isvals],
379:             &work[iu], &ldu, &work[ivt], &ldvt, &work[iwspc], &i1, info));
380:     if (*info) SETERRQ1(PETSC_COMM_SELF,1,"dsbtdc: Error in DGESVD, info = %d",*info);

382:     /* Note that after the return from DGESVD U is stored in */
383:     /* E(*,*,K), and V^{\top} is stored in WORK(IVT, IVT+1, ....) */

385:     /* determine the ranks RANK() for the approximations */

387:     rk = PetscMin(ksk,kskp1);
388: L8:
389:     dropsv = work[isvals - 1 + rk];

391:     if (dropsv * 2. <= tau1) {

393:       /* the error caused by dropping singular value RK is */
394:       /* small enough, try to reduce the rank by one more */

396:       if (--rk > 0) goto L8;
397:       else iwork[k] = 0;
398:     } else {

400:       /* the error caused by dropping singular value RK is */
401:       /* too large already, RK is the rank required to achieve the */
402:       /* desired accuracy */

404:       iwork[k] = rk;
405:     }

407: /* ************************************************************************** */

409:     /* Store the first RANK(K) terms of the SVD of the current */
410:     /* off-diagonal block. */
411:     /* NOTE that here it is required that L1E, L2E >= 2*KMAX+1 in order */
412:     /* to have enough space for storing singular vectors and values up */
413:     /* to the full SVD of an off-diagonal block !!!! */

415:     /* u1-u_RANK(K) is already contained in E(:,1:RANK(K),K) (as a */
416:     /* result of the call of DGESVD !), the sigma1-sigmaK are to be */
417:     /* stored in E(1:RANK(K),RANK(K)+1,K),  and v1-v_RANK(K) are to be */
418:     /* stored in E(:,RANK(K)+2:2*RANK(K)+1,K) */

420:     rp1 = iwork[k];
421:     for (j = 0; j < iwork[k]; ++j) {

423:       /* store sigma_J in E(J,RANK(K)+1,K) */

425:       e[j + (rp1 + k*l2e)* l1e] = work[isvals + j];

427:       /* update maximum norm of subdiagonal blocks */

429:       if (e[j + (rp1 + k*l2e)*l1e] > emax) {
430:         emax = e[j + (rp1 + k*l2e)*l1e];
431:       }

433:       /* store v_J in E(:,RANK(K)+1+J,K) */
434:       /* (note that WORK contains V^{\top} and therefore */
435:       /* we need to read rowwise !) */

437:       for (i = 1; i <= ksk; ++i) {
438:         e[i-1 + (rp1+j+1 + k*l2e)*l1e] = work[ivt+j + (i-1)*ldvt];
439:       }
440:     }

442:   }

444:   /* Compute the maximum norm of diagonal blocks and store the norm */
445:   /* of each diagonal block in E(RP1,RP1,K) (after the singular values); */
446:   /* store the norm of the last diagonal block in EXDNRM. */

448:   /* DMAX .... maximum one-norm of the diagonal blocks */

450:   dmax = 0.;
451:   for (k = 0; k < nblks; ++k) {
452:     rp1 = iwork[k];

454:     /* compute the one-norm of diagonal block K */

456:     dnrm = LAPACKlansy_("1", "L", &ksizes[k], &d[k*l1d*l2d], &l1d, work);
457:     if (k+1 == nblks) exdnrm = dnrm;
458:     else e[rp1 + (rp1 + k*l2e)*l1e] = dnrm;
459:     if (dnrm > dmax) dmax = dnrm;
460:   }

462:   /* Check for zero matrix. */

464:   if (emax == 0. && dmax == 0.) {
465:     for (i = 0; i < n; ++i) ev[i] = 0.;
466:     *info = -100;
467:     return(0);
468:   }

470: /* **************************************************************** */

472:   /* ....Identify irreducible parts of the block tridiagonal matrix */
473:   /* [while (START <= NBLKS)].... */

475:   start = 0;
476:   np = 0;
477: L10:
478:   if (start < nblks) {

480:     /* Let IEND be the number of the next subdiagonal block such that */
481:     /* its RANK is 0 or IEND = NBLKS if no such subdiagonal exists. */
482:     /* The matrix identified by the elements between the diagonal block START */
483:     /* and the diagonal block IEND constitutes an independent (irreducible) */
484:     /* sub-problem. */

486:     iend = start;

488: L20:
489:     if (iend < nblks) {
490:       rk = iwork[iend];

492:       /* NOTE: if RANK(IEND).EQ.0 then decoupling happens due to */
493:       /*       reduced accuracy requirements ! (because in this case */
494:       /*       we would not merge the corresponding two diagonal blocks) */

496:       /* NOTE: seems like any combination may potentially happen: */
497:       /*       (i) RANK = 0 but no decoupling due to small norm of */
498:       /*           off-diagonal block (corresponding diagonal blocks */
499:       /*           also have small norm) as well as */
500:       /*       (ii) RANK > 0 but decoupling due to small norm of */
501:       /*           off-diagonal block (corresponding diagonal blocks */
502:       /*           have very large norm) */
503:       /*       case (i) is ruled out by checking for RANK = 0 above */
504:       /*       (we decide to decouple all the time when the rank */
505:       /*       of an off-diagonal block is zero, independently of */
506:       /*       the norms of the corresponding diagonal blocks. */

508:       if (rk > 0) {

510:         /* check for decoupling due to small norm of off-diagonal block */
511:         /* (relative to the norms of the corresponding diagonal blocks) */

513:         if (iend == nblks-2) {
514:           d2 = PetscSqrtReal(exdnrm);
515:         } else {
516:           d2 = PetscSqrtReal(e[iwork[iend+1] + (iwork[iend+1] + (iend+1)*l2e)*l1e]);
517:         }

519:         /* this definition of TINY is analogous to the definition */
520:         /* in the tridiagonal divide&conquer (dstedc) */

522:         tiny = eps * PetscSqrtReal(e[iwork[iend] + (iwork[iend] + iend*l2e)*l1e])*d2;
523:         if (e[(iwork[iend] + iend*l2e)*l1e] > tiny) {

525:           /* no decoupling due to small norm of off-diagonal block */

527:           ++iend;
528:           goto L20;
529:         }
530:       }
531:     }

533:     /* ....(Sub) Problem determined: between diagonal blocks */
534:     /*     START and IEND. Compute its size and solve it.... */

536:     nrblks = iend - start + 1;
537:     if (nrblks == 1) {

539:       /* Isolated problem is a single diagonal block */

541:       nk = ksizes[start];

543:       /* copy this isolated block into Z */

545:       for (i = 0; i < nk; ++i) {
546:         ip = np + i + 1;
547:         for (j = 0; j <= i; ++j) z[ip + (np+j+1)*ldz] = d[i + (j + start*l2d)*l1d];
548:       }

550:       /* check whether there is enough workspace */

552:       spneed = 2*nk*nk + nk * 6 + 1;
553:       if (spneed > lwork) SETERRQ1(PETSC_COMM_SELF,1,"dsbtdc: not enough workspace for DSYEVD, info = %d",lwork - 200 - spneed);

555:       PetscStackCallBLAS("LAPACKsyevd",LAPACKsyevd_("V", "L", &nk,
556:                     &z[np + np*ldz], &ldz, &ev[np],
557:                     work, &lwork, &iwork[nblks-1], &liwork, info));
558:       if (*info) SETERRQ1(PETSC_COMM_SELF,1,"dsbtdc: Error in DSYEVD, info = %d",*info);
559:       start = iend + 1;
560:       np += nk;

562:       /* go to the next irreducible subproblem */

564:       goto L10;
565:     }

567:     /* ....Isolated problem consists of more than one diagonal block. */
568:     /*     Start the divide and conquer algorithm.... */

570:     /* Scale: Divide by the maximum of all norms of diagonal blocks */
571:     /*        and singular values of the subdiagonal blocks */

573:     /* ....determine maximum of the norms of all diagonal and subdiagonal */
574:     /*     blocks.... */

576:     if (iend == nblks-1) anorm = exdnrm;
577:     else anorm = e[iwork[iend] + (iwork[iend] + iend*l2e)*l1e];
578:     for (k = start; k < iend; ++k) {
579:       rp1 = iwork[k];

581:       /* norm of diagonal block */
582:       anorm = PetscMax(anorm,e[rp1 + (rp1 + k*l2e)*l1e]);

584:       /* singular value of subdiagonal block */
585:       anorm = PetscMax(anorm,e[(rp1 + k*l2e)*l1e]);
586:     }

588:     nk = 0;
589:     for (k = start; k < iend+1; ++k) {
590:       ksk = ksizes[k];
591:       nk += ksk;

593:       /* scale the diagonal block */
594:       PetscStackCallBLAS("LAPACKlascl",LAPACKlascl_("L", &zero, &zero,
595:                     &anorm, &done, &ksk, &ksk, &d[k*l2d*l1d], &l1d, info));
596:       if (*info) SETERRQ1(PETSC_COMM_SELF,1,"dsbtdc: Error in DLASCL, info = %d",*info);

598:       /* scale the (approximated) off-diagonal block by dividing its */
599:       /* singular values */

601:       if (k != iend) {

603:         /* the last subdiagonal block has index IEND-1 !!!! */
604:         for (i = 0; i < iwork[k]; ++i) {
605:           e[i + (iwork[k] + k*l2e)*l1e] /= anorm;
606:         }
607:       }
608:     }

610:     /* call the block-tridiagonal divide-and-conquer on the */
611:     /* irreducible subproblem which has been identified */

613:     BDC_dibtdc_(jobz, nk, nrblks, &ksizes[start], &d[start*l1d*l2d], l1d, l2d,
614:                 &e[start*l2e*l1e], &iwork[start], l1e, l2e, tau2, &ev[np],
615:                 &z[np + np*ldz], ldz, work, lwork, &iwork[nblks-1], liwork, info, 1);
616:                 
617:     if (*info) SETERRQ1(PETSC_COMM_SELF,1,"dsbtdc: Error in DIBTDC, info = %d",*info);

619: /* ************************************************************************** */

621:     /* Scale back the computed eigenvalues. */

623:     PetscStackCallBLAS("LAPACKlascl",LAPACKlascl_("G", &zero, &zero, &done,
624:             &anorm, &nk, &one, &ev[np], &nk, info));
625:     if (*info) SETERRQ1(PETSC_COMM_SELF,1,"dsbtdc: Error in DLASCL, info = %d",*info);

627:     start = iend + 1;
628:     np += nk;

630:     /* Go to the next irreducible subproblem. */

632:     goto L10;
633:   }

635:   /* ....If the problem split any number of times, then the eigenvalues */
636:   /* will not be properly ordered. Here we permute the eigenvalues */
637:   /* (and the associated eigenvectors) across the irreducible parts */
638:   /* into ascending order.... */

640:   /*  IF(NRBLKS.LT.NBLKS)THEN */

642:   /*    Use Selection Sort to minimize swaps of eigenvectors */

644:   for (ii = 1; ii < n; ++ii) {
645:     i = ii;
646:     k = i;
647:     p = ev[i];
648:     for (j = ii; j < n; ++j) {
649:       if (ev[j] < p) {
650:         k = j;
651:         p = ev[j];
652:       }
653:     }
654:     if (k != i) {
655:       ev[k] = ev[i];
656:       ev[i] = p;
657:       PetscStackCallBLAS("BLASswap",BLASswap_(&n, &z[i*ldz], &one, &z[k*ldz], &one));
658:     }
659:   }

661:   /* ...Compute MINGAP (minimum difference between neighboring eigenvalue */
662:   /*    approximations).............................................. */

664:   *mingap = ev[1] - ev[0];
665:   if (*mingap < 0.) SETERRQ2(PETSC_COMM_SELF,1,"dsbtdc: Eigenvalue approximations are not ordered properly. Approximation %d is larger than approximation %d.",1,2);
666:   *mingapi = 1;
667:   for (i = 2; i < n; ++i) {
668:     absdiff = ev[i] - ev[i-1];
669:     if (absdiff < 0.) SETERRQ2(PETSC_COMM_SELF,1,"dsbtdc: Eigenvalue approximations are not ordered properly. Approximation %d is larger than approximation %d.",i,i+1);
670:     else if (absdiff < *mingap) {
671:       *mingap = absdiff;
672:       *mingapi = i;
673:     }
674:   }

676:   /* check whether the minimum gap between eigenvalue approximations */
677:   /* may indicate severe inaccuracies in the eigenvector approximations */

679:   if (*mingap <= tol / 10) *info = -103;
680:   return(0);
681: }
