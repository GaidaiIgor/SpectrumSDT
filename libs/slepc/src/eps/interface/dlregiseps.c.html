 25: /*@C
 26:   EPSFinalizePackage - This function destroys everything in the SLEPc interface
 27:   to the EPS package. It is called from SlepcFinalize().

 29:   Level: developer

 31: .seealso: SlepcFinalize()
 32: @*/
 33: PetscErrorCode EPSFinalizePackage(void)
 34: {

 38:   PetscFunctionListDestroy(&EPSList);
 39:   EPSPackageInitialized = PETSC_FALSE;
 40:   EPSRegisterAllCalled  = PETSC_FALSE;
 41:   return(0);
 42: }

 44: /*@C
 45:   EPSInitializePackage - This function initializes everything in the EPS package.
 46:   It is called from PetscDLLibraryRegister() when using dynamic libraries, and
 47:   on the first call to EPSCreate() when using static libraries.

 49:   Level: developer

 51: .seealso: SlepcInitialize()
 52: @*/
 53: PetscErrorCode EPSInitializePackage()
 54: {
 55:   char           logList[256];
 56:   PetscBool      opt,pkg;
 57:   PetscClassId   classids[1];

 61:   if (EPSPackageInitialized) return(0);
 62:   EPSPackageInitialized = PETSC_TRUE;
 63:   /* Register Classes */
 64:   PetscClassIdRegister("EPS Solver",&EPS_CLASSID);
 65:   /* Register Constructors */
 66:   EPSRegisterAll();
 67:   /* Register Events */
 68:   PetscLogEventRegister("EPSSetUp",EPS_CLASSID,&EPS_SetUp);
 69:   PetscLogEventRegister("EPSSolve",EPS_CLASSID,&EPS_Solve);
 70:   /* Process Info */
 71:   classids[0] = EPS_CLASSID;
 72:   PetscInfoProcessClass("eps",1,&classids[0]);
 73:   /* Process summary exclusions */
 74:   PetscOptionsGetString(NULL,NULL,"-log_exclude",logList,sizeof(logList),&opt);
 75:   if (opt) {
 76:     PetscStrInList("eps",logList,',',&pkg);
 77:     if (pkg) { PetscLogEventDeactivateClass(EPS_CLASSID); }
 78:   }
 79:   /* Register package finalizer */
 80:   PetscRegisterFinalize(EPSFinalizePackage);
 81:   return(0);
 82: }

 84: #if defined(PETSC_HAVE_DYNAMIC_LIBRARIES)
 85: /*
 86:   PetscDLLibraryRegister - This function is called when the dynamic library

 90:   library.
 91:  */
 92: SLEPC_EXTERN PetscErrorCode PetscDLLibraryRegister_slepceps()
 93: {

 97:   EPSInitializePackage();
